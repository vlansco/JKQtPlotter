70  case 0: rgb= qRgba(val, qGreen(rgb), qBlue(rgb), qAlpha(rgb)); break;

287  /*! \brief convert a 2D image (as 1D array) into a QImage with given palette (see JKQTFPColorPalette)
288 
289  \param dbl_in pointer to a 1D array of template type \c T representing the image to plot. This array has to be of size \a width * \a height
290  \param width width of the array in \a dbl
291  \param height height of the array in \a dbl
292  \param[out] img the QImage object to draw to (should be initialized as \c QImage::Format_ARGB32 )
293  \param palette the color palette to use for the display
294  \param minColor lower boundary of color range in \a dbl pixels, if \a minColor == \a maxColor then this function will extract the image min and image max.
295  \param maxColor upper boundary of color range in \a dbl pixels, if \a minColor == \a maxColor then this function will extract the image min and image max.
296  \param paletteMinFail specifies what shell happen, when a value in \a dbl is below \a minColor
297  \param paletteMaxFail specifies what shell happen, when a value in \a dbl is above \a maxColor
298  \param minFailColor color to use for pixels that are below \a minColor for some settings of \a paletteMinFail
299  \param maxFailColor color to use for pixels that are below \a maxColor for some settings of \a paletteMaxFail
300  \param nanColor color to use for pixels that are not-a-number
301  \param infColor color to use for pixels that are infinity
302  \param logScale create a log-scaled image
303  \param logBase base for the logarithm used when \c logScale==true
304  \param lutUser user define LUT, used if \a palette \c ==JKQTPMathImageUSER_PALETTE
305  \note There is a variant of this function that is called with a userLUT directly, instead of \a palette
306  */
307  template <class T>
308  static inline void array2image(const T* dbl_in, int width, int height, QImage &img, JKQTPMathImageColorPalette palette, double minColor, double maxColor, JKQTPMathImageColorRangeFailAction paletteMinFail=JKQTPMathImageLastPaletteColor, JKQTPMathImageColorRangeFailAction paletteMaxFail=JKQTPMathImageLastPaletteColor, QColor minFailColor=QColor("black"), QColor maxFailColor=QColor("black"), QColor nanColor=QColor("black"), QColor infColor=QColor("black"), bool logScale=false, double logBase=10.0, const LUTType& lutUser=LUTType())
309  {
310  if (!dbl_in || width<=0 || height<=0)
311  return;
312 
313  double min = *dbl_in;
314  double max = *dbl_in;
315  if (jkqtp_approximatelyEqual(minColor, maxColor, JKQTP_DOUBLE_EPSILON)) {
316  bool first=true;
317  for (int i=1; i<width*height; ++i)
318  {
319  T v=dbl_in[i];
320  if (!(std::isnan(static_cast<long double>(v)) || std::isinf(static_cast<long double>(v)))) {
321  if (first) {
322  min=max=v;
323  first=false;
324  } else {
325  if (v < min)
326  min = v;
327  else if (v > max)
328  max = v;
329  }
330  }
331  }
332 
333  } else {
334  min = minColor;
335  max = maxColor;
336  }
337 
338  const T* dbl=dbl_in;
339  QVector<T> dbl1;
340  if (logScale) {
341  double logB=log10(logBase);
342  dbl1=QVector<T>(width*height, 0);
343  for (int i=0; i<width*height; i++) {
344  dbl1[i]=log10(dbl_in[i])/logB;
345  }
346  dbl=dbl1.data();
347  min=log10(min)/logB;
348  max=log10(max)/logB;
349  }
350  double delta=max-min;
351 
352 
353  // either use lutUser (if palette==JKQTPMathImageUSER_PALETTE) or else read the LUT from the internal storage
354  const LUTType& lut_used=(palette==JKQTPMathImageUSER_PALETTE)?lutUser:getLUTforPalette(JKQTPImageTools::global_jkqtpimagetools_lutstore, palette);
355 
356  img = QImage(width, height, QImage::Format_ARGB32);
357  if (jkqtp_approximatelyEqual(min, max, JKQTP_DOUBLE_EPSILON)) {
358  img.fill(0);
359  } else {
360  const int lutSize=lut_used.size();
361  if (lutSize>0) {
362  // LUT found: collor the image accordingly
363  for (int j=0; j<height; ++j) {
364  QRgb* line=reinterpret_cast<QRgb*>(img.scanLine(height-1-j));
365  for (int i=0; i<width; ++i) {
366  double val=dbl[j*width+i];
367  if (std::isnan(val)) {
368  line[i]=nanColor.rgba();
369  } else if (std::isinf(val)) {
370  line[i]=infColor.rgba();
371  } else {
372  const int v = static_cast<int>((val-min)/delta*static_cast<double>(lutSize-1));
373  const int vv = (v < 0) ? 0 : ( (v >= lutSize) ? (lutSize-1) : v);
374  line[i]=lut_used[vv];
375  if ((v<0)&&(paletteMinFail==JKQTPMathImageGivenColor)) {
376  line[i]=minFailColor.rgba();
377  } else if ((v>lutSize)&&(paletteMaxFail==JKQTPMathImageGivenColor)) {
378  line[i]=maxFailColor.rgba();
379  } else if ((v<0)&&(paletteMinFail==JKQTPMathImageTransparent)) {
380  line[i]=QColor(Qt::transparent).rgba();
381  } else if ((v>lutSize)&&(paletteMaxFail==JKQTPMathImageTransparent)) {
382  line[i]=QColor(Qt::transparent).rgba();
383  }
384  }
385  }
386  }
387  } else {
388  // no LUT found: paint a black image!
389  img.fill(0);
390  }
391  }
