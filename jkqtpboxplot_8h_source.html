38 /*! \brief This implements vertical <a href="http://en.wikipedia.org/wiki/Box_plot">boxplots</a>, optionally also a notched boxplot
39  \ingroup jkqtplotter_statgraphs
40 
41  The x position is given in posColumn. All other data are given in the medianColumn, minColumn, maxColumn,
42  percentile25Column and percentile75Column.
43 
44  \image html plot_boxplotvertical.png
45 
46 
47  The different features of a boxplot are:
48 
49  \image html plot_boxplotverticalelement.png
50 
51 
52  The example \ref JKQTPlotterBoxplotStyling discusses different options to style boxplots:
53 
54  \image html test_styledboxplot.png
55 
56  This class also implements hitTest() in a way that displays all data of the boxplot in the tooltips:
57 
58  \image html tooltip_boxplot.png
59 
60  \note There are additional classes to draw a single boxplot element: JKQTPBoxplotHorizontalElement and JKQTPBoxplotVerticalElement.
61  In these you can set the data values, as they are NOT drawn from a data column. This can be useful, if you e.g. want to
62  draw the statistical properties of a distribution.
63 
64  \section BoxplotOutliers Outliers
65 
66  If you also want to display outliers, as shown here with circles:
67 
68  \image html boxplot_outliers.png
69 
70  You need to add them as (x,y)-pairs to the datastore and add a separate JKQTPXYLineGraph that shows these. See \ref JKQTPlotterBoxplotsGraphs for details. Here is an example code-snippet:
71 
72  \code
73  // 4. create a graph of vertical boxplots:
74  JKQTPBoxplotVerticalGraph* graph=new JKQTPBoxplotVerticalGraph(&plot);
75  graph->setPositionColumn(columnPOS);
76  graph->setMinColumn(columnMIN);
77  graph->setPercentile25Column(columnQ25);
78  graph->setMedianColumn(columnMEDIAN);
79  graph->setMeanColumn(columnMEAN);
80  graph->setPercentile75Column(columnQ75);
81  graph->setMaxColumn(columnMAX);
82  graph->setTitle("vertical Boxplots");
83 
84  // 5. outliers need to be drawn separately
85  JKQTPXYLineGraph* graphOutliers=new JKQTPXYLineGraph(&plot);
86  graphOutliers->setXColumn(columnOUTLIERSX);
87  graphOutliers->setYColumn(columnOUTLIERSY);
88  graphOutliers->setTitle("outliers");
89  // make the color a darker shade of the color of graph
90  graphOutliers->setColor(graph->getColor().darker());
91  graphOutliers->setFillColor(QColor("white"));
92  // draw outliers as small circles, without lines
93  graphOutliers->setSymbolType(JKQTPCircle);
94  graphOutliers->setDrawLine(false);
95  graphOutliers->setSymbolSize(7);
96  \endcode
97 
98  \see \ref JKQTPlotterBoxplotsGraphs, jkqtpstatVAddBoxplots(),\ref JKQTPlotterBasicJKQTPDatastoreStatisticsGroupedStat, \ref JKQTPlotterBasicJKQTPDatastoreStatistics, \ref JKQTPlotterBoxplotStyling
99 
100  */
