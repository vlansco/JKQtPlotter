47 /** \brief adaptive drawing of a function graph, specified by two function \f$ f_x(t) \f$ and \f$ f_y(t) \f$ evaluated over a parameter range \f$ t\in\left[t_\text{min}..t_\text{max}\tight] \f$ */
48 class JKQTCOMMON_LIB_EXPORT JKQTPAdaptiveFunctionGraphEvaluator {
49 public:
50 
51  /** \brief class constructor
52  *
53  * \param fx function \f$ f_x(t) \f$
54  * \param fy function \f$ f_y(t) \f$
55  * \param minSamples the minimum number of points to evaluate the function at
56  * \param maxRefinementDegree the maximum number of recursive refinement steps
57  * each step bisects the interval \f$ [a, b] \f$ into two halfes. So the maximum number
58  * of points plotted at all are thus:
59  * \f[ \mbox{minSamples} \cdot 2^{\mbox{maxRefinementDegree}} \f]
60  * \param slopeTolerance the tolerance for the difference of two subsequent slopes
61  * \param minPixelPerSample create one sample at least every \a minPixelPerSample pixels
62  */
63  JKQTPAdaptiveFunctionGraphEvaluator(const std::function<double(double)>& fx_, const std::function<double(double)>& fy_, unsigned int minSamples_=10, unsigned int maxRefinementDegree_=5, double slopeTolerance_=0.005, double minPixelPerSample_=32);

65  /** \brief class constructor

67  * \param fxy function \f$ [x,y]=f_{xy}(t) \f$
68  * \param minSamples the minimum number of points to evaluate the function at
69  * \param maxRefinementDegree the maximum number of recursive refinement steps
70  * each step bisects the interval \f$ [a, b] \f$ into two halfes. So the maximum number
71  * of points plotted at all are thus:
72  * \f[ \mbox{minSamples} \cdot 2^{\mbox{maxRefinementDegree}} \f]
