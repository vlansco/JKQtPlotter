287 template <class TPainter>
288 inline void JKQTPPlotSymbol(TPainter& painter, double x, double y, JKQTPGraphSymbols symbol, double symbolSize, double symbolLineWidth, QColor color, QColor fillColor) {
289  painter.save(); auto __finalpaint=JKQTPFinally([&painter]() {painter.restore();});
290  QPen p=painter.pen();
291  p.setColor(color);
292  p.setWidthF(qMax(JKQTPlotterDrawingTools::ABS_MIN_LINEWIDTH, symbolLineWidth));
293  p.setStyle(Qt::SolidLine);
294  p.setCapStyle(Qt::FlatCap);
295  painter.setPen(p);
296  QBrush b=painter.brush();
297  b.setColor(fillColor);
298  b.setStyle(Qt::SolidPattern);
299  const double w=symbolSize;
300  const double w2=w/2.0;
301  const double w45=fabs(w*cos(45.0/180.0*JKQTPSTATISTICS_PI));
302  const double w3=w/3.0;
303 
304  // calculate star cordinates as static values
305  static int star5_items=0;
306  static double star5cordsx[10];
307  static double star5cordsy[10];
308  if (star5_items==0) {
309  star5_items=5;
310  double angle=360.0/double(star5_items)/180.0*JKQTPSTATISTICS_PI;
311  for (int i=0; i<star5_items; i++) {
312  double a=(static_cast<double>(i)+0.5)*angle;
313  star5cordsx[i*2]=sin(a);
314  star5cordsx[i*2+1]=0.5*sin(a+angle/2.0);
315  star5cordsy[i*2]=cos(a);
316  star5cordsy[i*2+1]=0.5*cos(a+angle/2.0);
317  }
318  }
319  static int star6_items=0;
320  static double star6cordsx[12];
321  static double star6cordsy[12];
322  if (star6_items==0) {
323  star6_items=6;
324  double angle=360.0/double(star6_items)/180.0*JKQTPSTATISTICS_PI;
325  for (int i=0; i<star6_items; i++) {
326  double a=(static_cast<double>(i)+0.5)*angle;
327  star6cordsx[i*2]=sin(a);
328  star6cordsx[i*2+1]=0.5*sin(a+angle/2.0);
329  star6cordsy[i*2]=cos(a);
330  star6cordsy[i*2+1]=0.5*cos(a+angle/2.0);
331  }
332  }
333  static int star8_items=0;
334  static double star8cordsx[16];
335  static double star8cordsy[16];
336  if (star8_items==0) {
337  star8_items=8;
338  double angle=360.0/double(star8_items)/180.0*JKQTPSTATISTICS_PI;
339  for (int i=0; i<star8_items; i++) {
340  double a=(static_cast<double>(i)+0.5)*angle;
341  star8cordsx[i*2]=sin(a);
342  star8cordsx[i*2+1]=0.5*sin(a+angle/2.0);
343  star8cordsy[i*2]=cos(a);
344  star8cordsy[i*2+1]=0.5*cos(a+angle/2.0);
345  }
346  }
347 
348  switch(symbol) {
349  case JKQTPDot:
350  painter.drawPoint(QPointF(x,y));
351  break;
352  case JKQTPCross:{
353  QPainterPath path;
354  path.moveTo(x-w2,y-w2);
355  path.lineTo(x+w2,y+w2);
356  path.moveTo(x-w2,y+w2);
357  path.lineTo(x+w2,y-w2);
358  painter.drawPath(path);
359  }
360  break;
361  case JKQTPPlus:{
362  QVector<QLineF> lines;
363  lines<<QLineF(x,y-w2,x,y+w2);
364  lines<<QLineF(x-w2,y,x+w2,y);
365  painter.drawLines(lines);
366  }
367  break;
368  case JKQTPCircle:{
369  QRectF rectangle(x-w2, y-w2, w, w);
370  painter.drawEllipse(rectangle);
371  }
372  break;
373  case JKQTPCircleCross:{
374  QRectF rectangle(x-w2, y-w2, w, w);
375  painter.drawEllipse(rectangle);
376  QVector<QLineF> lines;
377  lines<<QLineF(x-w45/2.0,y-w45/2.0,x+w45/2.0,y+w45/2.0);
378  lines<<QLineF(x-w45/2.0,y+w45/2.0,x+w45/2.0,y-w45/2.0);
379  painter.drawLines(lines);
380  }
381  break;
382  case JKQTPCirclePlus:{
383  QRectF rectangle(x-w2, y-w2, w, w);
384  painter.drawEllipse(rectangle);
385  QVector<QLineF> lines;
386  lines<<QLineF(x,y-w2,x,y+w2);
387  lines<<QLineF(x-w2,y,x+w2,y);
388  painter.drawLines(lines);
389  }
