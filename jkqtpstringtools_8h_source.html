54 inline QString JKQTPCDoubleToQString(double value) {
55  QLocale loc=QLocale::c();
56  loc.setNumberOptions(QLocale::OmitGroupSeparator);
57  return loc.toString(value, 'g', 18);
58 }

60 /** \brief convert a double to a string
61  * \ingroup jkqtptools_string
62  */
63 inline QString JKQTPDoubleToQString(double value, int prec = 10, char f = 'g', QChar decimalSeparator='.') {
64  QLocale loc=QLocale::c();
65  loc.setNumberOptions(QLocale::OmitGroupSeparator);
66  QString res=loc.toString(value, f, prec);
67  if (loc.decimalPoint()!=decimalSeparator) {
68  res=res.replace(loc.decimalPoint(), decimalSeparator);
69  }
70  return res;
71 }
