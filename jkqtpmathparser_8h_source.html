57 /*! \brief A simple function parser to parse (build memory tree representation) and
58  evaluate simple mathematical expressions
59  \ingroup jkmp
60 
61 
62  This class implements a simple function parser which can parse
63  mathematical expressions like <code> z=a*3+2.34^2*sin(pi*sqrt(x))</code> .
64  More than one expression can be separated by semicolon ';'. The result of
65  a parse operation will be a memory structure (tree) representing the given
66  expression.
67  The parser can cope with constants and (user defined) variables and supports
68  the data types number (double precision floating point), boolean (true/false)
69  and string. It already contains a lot of fundamental mathematical functions
70  (i.e. nearly all the functions from C StdLib).
71 
72  \section jkmp_constantsAndVars constants and variables:
73  This class provides management utilities for constants and variables. Variables
74  can also be defined as external variables, when a pointer to corresponding
75  memory is provided by the calling program. The parser supports a variable
76  assign operation \code a=<expression>\endcode which allows to define new
77  variables during evaluation. There are some mathematical standard constants
78  registered by calling JKQTPMathParser::addStandardVariables():
79  - \c pi = \f$ \pi \f$
80  - \c e = \f$ \exp(1) \f$
81  - \c sqrt2 = \f$ \sqrt{2} \f$
82  - \c version = the parser version
83  - \c log2e = \f$ \log_2(e) \f$
84  - \c log10e = \f$ \log_{10}(e) \f$
85  - \c ln2 = \f$ \ln(2)=\log_{e}(2) \f$
86  - \c ln10 = \f$ \ln(10)=\log_{e}(10) \f$
87  - \c h = \f$ 6.6260689633\cdot 10^{-34}\;\mathrm{J\cdot s} \f$ (planck constant)
88  - \c hbar = \f$ \hbar=\frac{h}{2\pi}= 1.05457162853\cdot 10^{-34}\;\mathrm{J\cdot s} \f$ (planck constant)
89  - \c epsilon0 = \f$ \epsilon_0= 8.854187817\cdot 10^{-12}\;\mathrm{\frac{F}{m}} \f$ (electric constant)
90  - \c mu0 = \f$ \mu_0=2\pi\cdot 10^{-7}= 12.566370614\cdot 10^{-7}\;\mathrm{\frac{N}{A^2}} \f$ (magnetic constant)
91  - \c c = \f$ 299792458\;\mathrm{\frac{m}{s}} \f$ (speed of light in vacuum)
92  - \c ce = \f$ 1.60217648740\cdot 10^{-19}\;\mathrm{C}\f$ (elementary charge)
93  - \c muB = \f$ \mu_B= 927.40091523\cdot 10^{-26}\;\mathrm{\frac{J}{T}} \f$ (Bohr magneton)
94  - \c muB_eV = \f$ \mu_B=5.788381755579\cdot 10^{-5}\;\mathrm{\frac{eV}{T}} \f$ (Bohr magneton)
95  - \c muN = \f$ \mu_N=5.0507832413\cdot 10^{-27}\;\mathrm{\frac{J}{T}} \f$ (nuclear magneton)
96  - \c muN_eV = \f$ \mu_N=3.152451232645\cdot 10^{-8}\;\mathrm{\frac{eV}{T}} \f$ (nuclear magneton)
97  - \c me = \f$ m_e= 9.1093821545\cdot 10^{-31}\;\mathrm{kg} \f$ (mass of electron)
98  - \c mp = \f$ m_p= 1.67262163783\cdot 10^{-27}\;\mathrm{kg} \f$ (mass of proton)
99  - \c mn = \f$ m_n= 1.67492721184\cdot 10^{-27}\;\mathrm{kg} \f$ (mass of neutron)
100  - \c NA = \f$ N_A= 6.0221417930\cdot 10^{23} \f$ (Avogadro constant = particles in 1 mol)
101  - \c kB = \f$ k_B=1.380650424\cdot 10^{-23}\;\mathrm{\frac{J}{K}} \f$ (Boltzman constant)
102  - \c kB_eV = \f$ k_B=8.61734315\cdot 10^{-5}\;\mathrm{\frac{eV}{K}} \f$ (Boltzman constant)
103 .
104  You can add user-defined contants by calling JKQTPMathParser::addVariableDouble()
105  JKQTPMathParser::addVariableBoolean() or JKQTPMathParser::addVariableString()

132  \section jkmp_resultsofparsing result of parsing and evaluation:
133  The result of calling JKQTPMathParser::parse()
134  will be a tree-like structure in memory. The parse() function will return
135  a pointer to the root node of this structure. All nodes inherit from
136  jkmpNode class. To evaluate such a structure simply call jkmpNode::evaluate()
137  of the root node. This will then return a jkmpResult structure which contains
138  the result. This scheme allows for once parsing and multiply evaluating an expression.
139  So if you want to define a function by an expression you can provide an
140  external variable x as the argument and then evaluate the function for
141  each x.
142 
143  \section jkmp_ebnf EBNF definition of the parsed expressions
144 
145 <pre> logical_expression -> logical_term
146  | logical_expression <b>or</b> logical_term
147  | logical_expression <b>||</b> logical_term</pre>
148 <pre> logical_term -> comp_expression
149  | logical_term <b>and</b> comp_expression
150  | logical_term <b>&amp;&amp;</b> comp_expression</pre>
151 <pre> comp_expression -> math_expression
152  | expression <b>==</b> math_expression
153  | expression <b>!=</b> math_expression
154  | expression <b>&gt;=</b> math_expression
155  | expression <b>&lt;=</b> math_expression
156  | expression <b>&gt;</b> math_expression
157  | expression <b>&lt;</b> math_expression</pre>
158 <pre> math_expression -> term
159  | math_expression <b>+</b> math_term
160  | math_expression <b>-</b> math_term</pre>
161 <pre> math_term -> primary
162  | term <b>*</b> primary
163  | term <b>/</b> primary
164  | term ( <b>%</b> | <b>mod</b> ) primary</pre>
165 <pre> primary -> <b>true</b> | <b>false</b>
166  | string_constant
167  | NUMBER
168  | NAME
169  | NAME <b>=</b> logical_expression
170  | <b>+</b> primary | <b>-</b> primary | <b>!</b> primary | <b>not</b> primary
171  | <b>(</b> logical_expression <b>)</b>
172  | NAME<b>(</b> parameter_list <b>)</b>
173  | primary <b>^</b> primary</pre>
174 <pre> string_constant -> <b>&quot;</b> STRING <b>&quot;</b> | <b>&apos;</b> STRING <b>&apos;</b></pre>
175 <pre> parameter_list -> \f$ \lambda \f$ | logical_expression | logical_expression <b>,</b> parameter_list</pre>

180  \section jkmp_example Simple Example of Usage
181  \code
182  try {
183  JKQTPMathParser mp; // instanciate
184  jkmpNode* n;
185  jkmpResult r;
186  // parse some numeric expression
187  n=mp.parse("pi^2+4*sin(65*pi/exp(3.45))");
188  r=n->evaluate();
189  cout<<r.num<<endl;
190  //delete n;
191 
192  // parse some boolean expression
193  n=mp.parse("true==false");
194  r=n->evaluate();
195  if (r.type==jkmpBool) {
196  if (r.boolean) cout<<"true";
197  else cout<<"false";
198  }
199  if (r.type==jkmpDouble) cout<<r.num<<endl;
200  if (r.type==jkmpString) cout<<r.str<<endl;
201  delete n;
202 
203  // parse some string expression
204  n=mp.parse("var1='false'; var1+'true'");
205  r=n->evaluate();
206  if (r.type==jkmpString) cout<<r.str<<endl;
207  delete n;
208  } catch(std::exception& E) {
209  cout<<"ERROR!!!\n "<<E.what()<<endl<<endl;
210  }
211  \endcode

361  /** \brief This is a function prototype for adding new mathematical functions
362  * to the parser
363  *
364  * If you want to add more math functions (like sin, cos , abs ...) to the
365  * parser, you will have to implement it with this prototype and then register
366  * it with JKQTPMathParser::addFunction(). The first parameter points to an array
367  * containing the input parameters while the second one specifies the number
368  * of supplied parameters. The result has to be of type jkmpResult.
369  *
370  * All error handling has to be done inside the function definition. Here is a
371  * simple example:
372  * \code
373  * jkmpResult Abs(jkmpResult* params, unsigned char n){
374  * jkmpResult r;
375  * r.type=jkmpDouble;
376  * if (n!=1) jkmpError("abs accepts 1 argument");
377  * if (params[0].type!=jkmpDouble) jkmpError("abs needs double argument");
378  * r.num=fabs(params[0].num);
379  * return r;
380  * }
381  * \endcode
382  */
383  typedef jkmpResult (*jkmpEvaluateFunc)(jkmpResult*, unsigned char, JKQTPMathParser*);
