47 /*! \brief fits (in a least-squares sense) a polynomial \f$ f(x)=\sum\limits_{i=0}^Pp_ix^i \f$ of order P to a set of N data pairs \f$ (x_i,y_i) \f$
48  \ingroup jkqtptools_math_statistics_poly
49 
50  \tparam InputItX standard iterator type of \a firstX and \a lastX.
51  \tparam InputItY standard iterator type of \a firstY and \a lastY.
52  \tparam OutputItP output iterator for the polynomial coefficients
53  \param type model to be fitted
54  \param firstX iterator pointing to the first item in the x-dataset to use \f$ x_1 \f$
55  \param lastX iterator pointing behind the last item in the x-dataset to use \f$ x_N \f$
56  \param firstY iterator pointing to the first item in the y-dataset to use \f$ y_1 \f$
57  \param lastY iterator pointing behind the last item in the y-dataset to use \f$ y_N \f$
58  \param P degree of the polynomial (P>=N !!!)
59  \param[out] firstRes Iterator (of type \a OutputItP ), which receives the (P+1)-entry vector with the polynomial coefficients \f$ p_i \f$
60 
61  This function uses jkqtpstatLinSolve() to solve the system of equations
62  \f[ \begin{bmatrix} y_1\\ y_2\\ y_3 \\ \vdots \\ y_n \end{bmatrix}= \begin{bmatrix} 1 & x_1 & x_1^2 & \dots & x_1^P \\ 1 & x_2 & x_2^2 & \dots & x_2^P\\ 1 & x_3 & x_3^2 & \dots & x_3^P \\ \vdots & \vdots & \vdots & & \vdots \\ 1 & x_n & x_n^2 & \dots & x_n^P \end{bmatrix} \begin{bmatrix} p_0\\ p_1\\ p_2\\ \vdots \\ p_P \end{bmatrix} \f]
63  \f[ \vec{y}=V\vec{p}\ \ \ \ \ \Rightarrow\ \ \ \ \ \vec{p}=(V^TV)^{-1}V^T\vec{y} \f]
64 
65  \image html datastore_regression_polynom.png
66 
67  \see https://en.wikipedia.org/wiki/Polynomial_regression
68 */
69 template <class InputItX, class InputItY, class OutputItP>
70 inline void jkqtpstatPolyFit(InputItX firstX, InputItX lastX, InputItY firstY, InputItY lastY, size_t P, OutputItP firstRes) {
71  {
72  const int Nx=std::distance(firstX,lastX);
73  const int Ny=std::distance(firstY,lastY);
74  JKQTPASSERT(Nx>1 && Ny>1);
75  }
76 
77  size_t N=0;
78 
79  std::vector<double> X,Y;
80  auto itX=firstX;
81  auto itY=firstY;
82  for (; itX!=lastX && itY!=lastY; ++itX, ++itY) {
83  const double fit_x=jkqtp_todouble(*itX);
84  const double fit_y=jkqtp_todouble(*itY);
85  if (JKQTPIsOKFloat(fit_x) && JKQTPIsOKFloat(fit_y)) {
86  X.push_back(fit_x);
87  Y.push_back(fit_y);
88  N++;
89  }
90  }
91 
92  // build Vandermonde matrix V
93  std::vector<double> V;
94  V.resize(N*(P+1));
95  for (size_t l=0; l<N; l++) {
96  V[jkqtplinalgMatIndex(l,0,P+1)]=1.0;
97  double x=X[l];
98  const double xx=x;
99  for (size_t c=1; c<P+1; c++) {
100  V[jkqtplinalgMatIndex(l,c,P+1)]=x;
101  x=x*xx;
102  }
103  }
104 #ifdef STATISTICS_TOOLS_DEBUG_statisticsPolyFit
105  std::cout<<"V = \n";
106  jkqtplinalgPrintMatrix(V.data(),N,P+1);
107  std::cout<<"\n";
108 #endif
109 
110  // calculate V^T
111  std::vector<double> VT=V;
112  jkqtplinalgTransposeMatrix(VT.data(), static_cast<long>(N), static_cast<long>(P+1));
113 
114 #ifdef STATISTICS_TOOLS_DEBUG_statisticsPolyFit
115  std::cout<<"V^T = \n";
116  jkqtplinalgPrintMatrix(VT.data(),P+1,N);
117  std::cout<<"\n";
118 #endif
119 
120  // calculate V^T*V
121  std::vector<double> VTV;
122  VTV.resize((P+1)*(P+1));
123  jkqtplinalgMatrixProduct(VT.data(), static_cast<long>(P+1), static_cast<long>(N), V.data(), static_cast<long>(N), static_cast<long>(P+1), VTV.data());
124 
125 #ifdef STATISTICS_TOOLS_DEBUG_statisticsPolyFit
126  std::cout<<"V^T*V = \n";
127  jkqtplinalgPrintMatrix(VTV.data(),P+1,P+1);
128  std::cout<<"\n";
129 #endif
130 
131  // calculate V^T*y
132  std::vector<double> VTY;
133  VTY.resize(P+1);
134  jkqtplinalgMatrixProduct(VT.data(), static_cast<long>(P+1), static_cast<long>(N), Y.data(), static_cast<long>(N), 1, VTY.data());
135 
136 #ifdef STATISTICS_TOOLS_DEBUG_statisticsPolyFit
137  std::cout<<"V^T*y = \n";
138  jkqtplinalgPrintMatrix(VTY.data(),P+1,1);
139  std::cout<<"\n";
140 #endif
141 
142  // solve V^T*y = V^T*V*p
143  const bool ok=jkqtplinalgLinSolve(VTV.data(), VTY.data(), static_cast<long>(P+1));
144 
145  if (ok) {
146  auto itR=firstRes;
147  for (size_t p=0; p<P+1; p++) {
148  *itR=VTY[p];
149  ++itR;
150  }
151  } else {
152  throw std::runtime_error("jkqtplinalgLinSolve() didn't return a result!");
153  }
154 
155 #ifdef STATISTICS_TOOLS_DEBUG_statisticsPolyFit
156  std::cout<<"result_out = \n";
157  jkqtplinalgPrintMatrix(result_out,P+1,1);
158  std::cout<<"\n";
159 #endif
160 
161 }
