456 /*! \brief evaluate a polynomial \f$ f(x)=\sum\limits_{i=0}^Pp_ix^i \f$ with \f$ p_i \f$ taken from the range \a firstP ... \a lastP
457  \ingroup jkqtptools_math_basic
458 
459  \tparam PolyItP iterator for the polynomial coefficients
460  \param x where to evaluate
461  \param firstP points to the first polynomial coefficient \f$ p_1 \f$ (i.e. the offset with \f$ x^0 \f$ )
462  \param lastP points behind the last polynomial coefficient \f$ p_P \f$
463  \return value of polynomial \f$ f(x)=\sum\limits_{i=0}^Pp_ix^i \f$ at location \a x
464 
465 */
466 template <class PolyItP>
467 inline double jkqtp_polyEval(double x, PolyItP firstP, PolyItP lastP) {
468  double v=0.0;
469  double xx=1.0;
470  for (auto itP=firstP; itP!=lastP; ++itP) {
471  v=v+(*itP)*xx;
472  xx=xx*x;
473  }
474  return v;
475 }

504 /*! \brief Generates a LaTeX string for the polynomial model with the coefficients \a firstP ... \a lastP
505  \ingroup jkqtptools_math_basic
506 
507  \tparam PolyItP iterator for the polynomial coefficients
508  \param firstP points to the first polynomial coefficient \f$ p_1 \f$ (i.e. the offset with \f$ x^0 \f$ )
509  \param lastP points behind the last polynomial coefficient \f$ p_P \f$
510  */
511 template <class PolyItP>
512 QString jkqtp_polynomialModel2Latex(PolyItP firstP, PolyItP lastP) {
513  QString str="f(x)=";
514  size_t p=0;
515  for (auto itP=firstP; itP!=lastP; ++itP) {
516  if (p==0) str+=jkqtp_floattolatexqstr(*itP, 3);
517  else {
518  if (*itP>=0) str+="+";
519  str+=QString("%2{\\cdot}x^{%1}").arg(p).arg(jkqtp_floattolatexqstr(*itP, 3));
520  }
521  p++;
522  }
523  return str;
524 }
